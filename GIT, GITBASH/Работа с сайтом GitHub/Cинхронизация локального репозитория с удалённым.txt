# находясь в папке с локальным репозиторием, привязываем его к удалённому (URL)
git remote add origin https://github.com/oleg-kuzmin/work.git

# заливаем все файлы из локального репозитория в удалённый, который уже привязали
git push -u origin main

# Команды для публикации изменений - забрать изменения, сделанные другими
git pull

# Команды для публикации изменений - опубликовать изменения в удалённый репозиторий
git push
