# По умолчанию директория с SSH-ключами находится в домашней директории пользователя:
cd ~

# Обычно SSH-ключи находятся в директории .ssh/ Вывести список созданных ключей:
ls -la .ssh/

# Для более структурированного вывода можно добавить пару параметров:
ls -lAFho .ssh/

# Если папка пустая или её нет, то всё хорошо.
# Если есть с похожим названием, SSH-ключи уже создавались:
•	id_dsa.pub
•	id_ecdsa.pub
•	id_ed25519.pub
•	id_rsa.pub

# Если вы не создавали их, удалите все эти файлы.

# Удаление ключа
rm -rf ~/.ssh
