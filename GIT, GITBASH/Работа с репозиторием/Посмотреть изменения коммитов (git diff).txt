# Посмотреть изменения коммитов (git diff)
# показывает только изменения в «рабочей зоне» — когда файлы ещё не добавлены в stage и не опубликованы.
git diff

#  посмотреть изменения, когда файлы добавлены в stage
git diff --staged

#  сравнить изменения двух конкретных коммитов
git diff a9928ab 11bada1

#  посмотреть изменения в index.html
diff index.html
diff --cc index.html
