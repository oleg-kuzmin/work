# Открывается при команде
git commit

# Войти в режим редактирования (внизу будет надпись ВСТАВКА)
нажать "i"

# Выход из режима просмотра/редактирования в командный режим (курсор внизу)
нажать "ESC"

# Выйти из файла сохранив изменения (командный режим, горячие клавиши)
ZZ (зажат Shift, поочередное нажатие)

# Выйти из файла сохранив изменения (командный режим, ввод в терминале)
:wq

# Выйти из файла без изменений (командный режим, горячие клавиши)
ZQ (зажат Shift, поочередное нажатие) — выход без сохранения

# Выйти из файла без изменений (командный режим, ввод в терминале)
:q!
