git add file-name

# Добавить все файлы для отслеживания (индексации) в staging area

git add --all   (добавит все файлы для индексирования)
git add -A      (добавит все файлы для индексирования)
git add .       (добавит все файлы для индексирования)

*****************************************************************
# Добавить файл частично для отслеживания (индексации) в staging area
# начать интерактивный сеанс индексирования, вы сможете выбрать части файла, которые будут добавлены в следующий коммит
# команда представит фрагмент изменений и предложит вам ввести команду

git add -P

y   # проиндексировать фрагмент
n   # игнорировать фрагмент
s   # разбить его на более мелкие фрагменты
e   # вручную отредактировать фрагмент
q   # завершить работу с командой
